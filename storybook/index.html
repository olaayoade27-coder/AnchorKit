💻 Quick Start

// Install AnchorKit
cargo add anchor-kit

// Configure the SDK
use anchor_kit::config::SdkConfig;

let config = SdkConfig::builder()
  .network(SdkNetwork::Testnet)
  .anchor_domain("demo.anchor.com")
  .build()
  .expect("Valid config");

// Create the client
let client = AnchorClient::new(config)
  .await   .expect("Client created");
